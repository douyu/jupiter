# 4.3.2 在 Juno 里启用 Gitlab

enabled = true
allow_sign_up = false

client_secret = "GITLAB_SECRET"
scopes = "api"
auth_url = "https://gitlab.com/oauth/authorize"

api_url = "https://gitlab.com/api/v4"
